(function(){

         function Builder(){

                   this.liNode = '';

         }

         Builder.prototype = {

                   addLi : function(text){

                            this.liNode += '<li>'+text+'</li>';

                   },

                   toString : function(){

                            return '<ul>'+this.liNode+'</ul>';

                   }

         }

         var c = new Builder();

         c.addLi('执行完成');

         c.addLi('注意到ul了么？');

         document.getElementById('temp').innerHTML = c;

})();
